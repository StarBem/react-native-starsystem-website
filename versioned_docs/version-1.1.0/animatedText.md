Declarative transitions and animations for React Native Text

## Usage

```js
import { View } from "react-native";
import { AnimatedText, Text } from "react-native-starsystem";

const Example = () => {
  return (
    <AnimatedText animation="fadeIn">
      <Text>Example</Text>
    </AnimatedText>
  );
};
```
