Declarative transitions and animations for React Native View

## Usage

```js
import { View } from "react-native";
import { AnimatedView } from "react-native-starsystem";

const Example = () => {
  return (
    <AnimatedView animation="fadeIn">
      <View />
    </AnimatedView>
  );
};
```
